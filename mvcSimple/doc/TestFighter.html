  }

  param2 ={ name: "SubZero",
           defense: 3, # > 0
           attack: 5,  # > 0
           life: 5 #cant be < 2
  }

  @scorpio = Fighter.new(param1[:name],param1[:defense],param1[:attack],param1[:life])
  @subzero = Fighter.new(param2[:name],param2[:defense],param2[:attack],param2[:life])
